Esempio - Tipi Primitivi

// string
let nome: string = "Marco";
let cognome: string = 'Rossi';
let fullName: string = `${nome} ${cognome}`;

// number
let eta: number = 25;
let prezzo: number = 19.99;
let hex: number = 0xFF;  // 255

// boolean
let attivo: boolean = true;
let verificato: boolean = false;

// any (evita!)
let qualsiasi: any = "ciao";
qualsiasi = 123;
qualsiasi = true;

// unknown (meglio di any)
let sconosciuto: unknown = "test";
if (typeof sconosciuto === "string") {
    console.log(sconosciuto.toUpperCase());
}

// void
function log(msg: string): void {
    console.log(msg);
}

// null e undefined
let nullo: null = null;
let indefinito: undefined = undefined;

// Union con null
let opzionale: string | null = "valore";
opzionale = null;

console.log("✅ Tutti i tipi primitivi funzionano!");
