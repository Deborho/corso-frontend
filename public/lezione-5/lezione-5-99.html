6. localStorage con JSON

// Salva
saveToStorage() {
    const jsonString = JSON.stringify(this.todos);
    localStorage.setItem(this.STORAGE_KEY, jsonString);
}

// Carica
loadFromStorage() {
    const jsonString = localStorage.getItem(this.STORAGE_KEY);
    return jsonString ? JSON.parse(jsonString) : [];
}

7. Event Delegation

// Un solo listener per tutti i todo
this.view.todoList.addEventListener('click', (e) => {
    const target = e.target;
    const id = parseInt(target.dataset.id);
    
    if (target.classList.contains('btn-delete')) {
        this.handleDeleteTodo(id);
    }
});

8. Async/Await & Promises

// Handler asincrono
async handleEditTodo(id) {
    const todo = this.model.getTodos().find(t => t.id === id);
    
    // await per modal con Promise
    const result = await this.view.showEditModal(todo);
    
    if (result.action === 'save') {
        this.model.updateTodo(id, result.text);
    }
}
