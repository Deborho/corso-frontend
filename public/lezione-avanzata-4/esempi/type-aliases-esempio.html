Esempio - Type Aliases e Literal Types

// Type aliases
type ID = string | number;
type Email = string;

let userId: ID = "U-001";
let adminId: ID = 42;

// Literal types
type Direzione = "nord" | "sud" | "est" | "ovest";
type HttpMethod = "GET" | "POST" | "PUT" | "DELETE";

function muovi(dir: Direzione) {
    console.log(`Muovo verso ${dir}`);
}

muovi("nord");  // ✅ OK
// muovi("left");  // ❌ Errore

// Template literal types
type Colore = "rosso" | "verde" | "blu";
type Tonalita = "chiaro" | "scuro";
type ColoreCompleto = `${Colore}-${Tonalita}`;

let colore: ColoreCompleto = "rosso-chiaro";

// Type per oggetti
type Utente = {
    id: ID;
    nome: string;
    email: Email;
    attivo: boolean;
};

let utente: Utente = {
    id: 1,
    nome: "Marco",
    email: "marco@email.com",
    attivo: true
};

// Type per funzioni
type Callback = (error: Error | null, data?: any) => void;
type Comparatore = (a: number, b: number) => number;

const crescente: Comparatore = (a, b) => a - b;

// Utility types
type UtenteParziale = Partial<Utente>;
let update: UtenteParziale = { email: "nuovo@email.com" };

type UtenteReadonly = Readonly<Utente>;
type UtenteMinimo = Pick<Utente, "id" | "nome">;
type UtenteSenzaID = Omit<Utente, "id">;

console.log("✅ Type aliases funzionano!");
