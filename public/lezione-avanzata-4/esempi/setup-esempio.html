Esempio Pratico - Setup TypeScript

Installazione e Configurazione

Vediamo come configurare un progetto TypeScript da zero.

1. Installazione TypeScript

# Installa TypeScript globalmente
npm install -g typescript

# Verifica installazione
tsc --version

# Oppure installa localmente nel progetto
npm init -y
npm install --save-dev typescript

2. Crea tsconfig.json

# Genera tsconfig.json con defaults
tsc --init

# Oppure crea manualmente:
{
  "compilerOptions": {
    "target": "ES2020",
    "module": "ESNext",
    "outDir": "./dist",
    "rootDir": "./src",
    "strict": true,
    "esModuleInterop": true,
    "skipLibCheck": true,
    "forceConsistentCasingInFileNames": true
  },
  "include": ["src/**/*"],
  "exclude": ["node_modules"]
}

3. Compila e Esegui

# Compila singolo file
tsc src/hello.ts

# Compila tutto il progetto
tsc

# Watch mode (ricompila automaticamente)
tsc --watch

# Esegui il JavaScript compilato
node dist/hello.js

üìù Esempio con Interface

interface Utente {
    id: number;
    nome: string;
    email: string;
    attivo?: boolean;
}

function creaUtente(
    nome: string,
    email: string
): Utente {
    return {
        id: Date.now(),
        nome,
        email,
        attivo: true
    };
}

const utente = creaUtente(
    "Marco Rossi",
    "marco@email.com"
);

console.log(utente.nome);  // Autocomplete!

4. Package.json Scripts

{
  "name": "mio-progetto-ts",
  "version": "1.0.0",
  "scripts": {
    "build": "tsc",
    "watch": "tsc --watch",
    "start": "node dist/index.js",
    "dev": "tsc && node dist/index.js"
  },
  "devDependencies": {
    "typescript": "^5.3.0"
  }
}
