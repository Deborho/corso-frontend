  const age = 25
  const isAdult = age >= 18
  
  return (

      {/* Variabili */}

    </div>
  )
}

Differenze JSX vs HTML

function JSXDifferences() {
  return (
    <div>
      {/* className invece di class */}
      <div className="container">Contenitore</div>
      
      {/* htmlFor invece di for */}
      <label htmlFor="email">Email:</label>
      <input id="email" type="email" />
      
      {/* camelCase per attributi */}
      <button onClick={() => alert('Click')}>Click me</button>
      <div tabIndex={0}>Focusable</div>
      
      {/* Self-closing tags devono chiudere */}
      <img src="photo.jpg" alt="Photo" />
      <input type="text" />
      <br />
      
      {/* style √® un oggetto JavaScript */}
      <div style={{ color: 'red', fontSize: '20px' }}>
        Styled text
      </div>
    </div>
  )
}
